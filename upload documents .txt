                <input type="file" id="idDocument" accept=".jpg, .jpeg, .png, .pdf" required>
            </div>

            <div class="form-group">
                <label for="matricCertificate">Matric Certificate / Statement of Results</label>
                <input type="file" id="matricCertificate" accept=".jpg, .jpeg, .png, .pdf" required>
            </div>

            <div class="form-group">
                <label for="grade11Report">Grade 11 Report</label>
                <input type="file" id="grade11Report" accept=".jpg, .jpeg, .png, .pdf" required>

        <div class="upload-result" id="uploadResult">
            <!-- This will display the uploaded file names and any success/error messages -->
        </div>
    </div>

    <script>
        // Handle file upload form submission
        document.getElementById("uploadForm").addEventListener("submit", function(e) {
            e.preventDefault();

            // Get file inputs
            const idDocument = document.getElementById("idDocument").files[0];
            const matricCertificate = document.getElementById("matricCertificate").files[0];
            const grade11Report = document.getElementById("grade11Report").files[0];

            // Check if all files are uploaded
            if (!idDocument || !matricCertificate || !grade11Report) {
                alert("Please upload all required documents.");
                return;
            }

            // Simulate file upload and display result
            const uploadResultDiv = document.getElementById("uploadResult");
            uploadResultDiv.innerHTML = `
                <h3>Documents Successfully Uploaded:</h3>
                <ul>
                    <li><strong>ID Document:</strong> ${idDocument.name}</li>
                    <li><strong>Matric Certificate:</strong> ${matricCertificate.name}</li>
                    <li><strong>Grade 11 Report:</strong> ${grade11Report.name}</li>
                </ul>
            `;
        });
    </script>
